
* 사진 게시판 */
CREATE TABLE PICS (
	PICS_NUM NUMBER NOT NULL, /* 사진 ID */
	PICS_MEMBER NUMBER, /* 글쓴이 */
	PICS_NICK VARCHAR2(50), /* 글쓴이별명 */
	PICS_DATE DATE, /* 작성일자 */
	PICS_CATEGORY VARCHAR2(30), /* 카테고리 */
	PICS_LIKE NUMBER, /* 좋아요 수 */
	PICS_WORKSHOP VARCHAR2(100), /* 제작 공방 */
	PICS_TAG VARCHAR2(100), /* 태그 이름 */
	PICS_READ NUMBER, /* 조회수 */
	PICS_REVIEW NUMBER, /* 후기 */
	PICS_CONTENT_1 VARCHAR2(2000), /* 내용1 */
	PICS_CONTENT_2 VARCHAR2(2000), /* 내용2 */
	PICS_CONTENT_3 VARCHAR2(2000), /* 내용3 */
	PICS_FILE_1 VARCHAR2(1000), /* 파일경로1 */
	PICS_FILE_2 VARCHAR2(1000), /* 파일경로2 */
	PICS_FILE_3 VARCHAR2(1000), /* 파일경로3 */
	PICS_REF NUMBER /* 파일그룹 */
);

CREATE SEQUENCE SEQ_PICS
    START WITH 1
    INCREMENT BY 1;

CREATE SEQUENCE SEQ_PICS_REF
    START WITH 1
    INCREMENT BY 1;
    
ALTER TABLE PICS
	ADD
		CONSTRAINT PK_PICS
		PRIMARY KEY (
			PICS_NUM
		);
